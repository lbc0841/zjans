> 給一矩陣 A<br>
對 A 沿著 主對角線 進行翻轉

***

## 解法一

### ✅ 完整代碼

評分結果(參考) ： **<font color="#00bb00">AC</font> (2ms, 340KB)**

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int m, n;
    while(cin >> m >> n){
        vector<vector<int>> arr(m+n, vector<int>(m+n));

        for(int i=0; i<m; i++){
            for(int j=0; j<n; j++){
                cin >> arr[i][j];
            }          
        }

        for(int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                if(j!=0) cout << " ";
                cout << arr[j][i];
            }
            cout << "\n";
        }
    }

    return 0;
}
```
