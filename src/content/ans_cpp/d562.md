## 解法一

### ✅ 完整代碼

評分結果(參考) ： **<font color="#00bb00">AC</font> (89ms, 320KB)**

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

signed main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    while(cin >> n){
        vector<int> a(n);
        for(int& i : a) cin >> i;
        
        bool rev = false;
        int l=0, r=n;
        for(int i=0; i<n; i++){
            for(int j=l; j<r; j++){
                if(rev) cout << a[n-j-1] << " ";
                else cout << a[j] << " ";
            }
            if(rev) l++;
            else r--;
            rev = !rev;
            cout << "\n";
        }
        cout << "\n";
    }
    
    return 0;
}
```
