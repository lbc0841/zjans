## 解法一

### ✅ 完整代碼

評分結果(參考) ： **<font color="#00bb00">AC</font> (1ms, 400KB)**

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

signed main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    vector<int> dp(10000);
    dp[0] = 1;
    dp[1] = 2;
    for(int i=2; i<10000; i++){
        dp[i] = (dp[i-1]+dp[i-2])%1000000007;
    }
    
    int n;
    while(cin >> n){
        cout << dp[n-1] << "\n";
    }
    
    return 0;
}
```
