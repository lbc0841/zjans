評分結果(參考) ： **<font color="#00bb00">AC</font> (2ms, 328KB)**

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int a, b;
    while(cin >> a >> b){
        if(a==-1 && b==-1) break;
        cout << min(abs(a-b), 100-abs(a-b)) << "\n";
    }

    return 0;
}
```
