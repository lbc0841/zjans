### ✅ 完整代碼

評分結果(參考) ： **<font color="#00bb00">AC</font> (2ms, 364KB)**

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, x, mx = 0;
    cin >> n;
    
    for(int i=0; i<n; i++){
        cin >> x;
        mx = max(mx, x);
    }
    cout << mx;
    
    return 0;
}
```
