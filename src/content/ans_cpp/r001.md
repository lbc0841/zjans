## 解法一

### ✅ 完整代碼

評分結果(參考) ： **<font color="#00bb00">AC</font> (7ms, 352KB)**

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int cnt = 0;
    for(int i=1; cnt<100; i++){
        int sum1=0, sum2=0;
        for(int j=i; j<i+cnt+2; j++) sum1 += j*j;
        for(int j=i+cnt+2; j<i+cnt+cnt+3; j++) sum2 += j*j;
        
        if(sum1 == sum2){
            for(int j=i; j<i+cnt+2; j++){
                if(j != i) cout << " + ";
                cout << j << "^2";
            }
            cout << " = ";
            for(int j=i+cnt+2; j<i+cnt+cnt+3; j++){
                if(j != i+cnt+2) cout << " + ";
                cout << j << "^2";
            }
            cout << "\n";
            cnt++;
        }
    }
    
    return 0;
}
```
