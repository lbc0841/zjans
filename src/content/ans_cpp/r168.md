## 解法一

### ✅ 完整代碼

評分結果(參考) ： **<font color="#00bb00">AC</font> (40ms, 1.3MB)**

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

signed main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    vector<vector<char>> v(n, vector<char>(n, '0'));
    for(int i=0; i<n; i++){
        for(int j=i; j<n; j++){
            if(m <= 0) break;
            
            if(i == j){
                v[i][j] = '1';
                m--;
            }
            else{
                if(m >= 2)
                {
                    v[i][j] = '1';
                    v[j][i] = '1';
                    m -= 2;
                }
            }
        }
    }
    
    for(auto& a : v){
        for(char c : a){
            cout << c << " ";
        }
        cout << "\n";
    }
    
    return 0;
}
```
