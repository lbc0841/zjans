```cpp
#include <bits/stdc++.h>
using namespace std;

inline double log_fact_stirling(long long n){
    if(n <= 1) return 0.0;
    return n * log10(n / M_E) + 0.5 * log10(2 * M_PI * n);
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long n, k;
    while (cin >> n >> k){
        if (n == k){
            cout << "1\n";
            continue;
        }
        
        double ans = log_fact_stirling(n) 
                   - log_fact_stirling(k) 
                   - log_fact_stirling(n - k);
        cout << (long long)ceil(ans) << "\n";
    }
    return 0;
}
```
