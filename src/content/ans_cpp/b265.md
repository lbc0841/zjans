#include <bits/stdc++.h>
using namespace std;
#define int long long

signed main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    vector<int> nums(5);
    
    while(true){
        cin >> n;
        if(n == 0) break;
        
        int mx=0, cnt=1;
        unordered_map<int, int> um;
        
        for(int i=0; i<n; i++){
            for(int& num : nums) cin >> num;
            sort(nums.begin(), nums.end());
            
            int k = nums[0] + (nums[1]<<9) + (nums[2]<<18) + (nums[3]<<27) + (nums[4]<<36);
            um[k]++;
            
            if(um[k] > mx){
                mx = um[k];
                cnt = 1;
            }
            else if(um[k] == mx){
                cnt++;
            }
        }
        cout << mx*cnt << "\n";
    }
    
    return 0;
}
