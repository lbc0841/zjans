### ✅ 完整代碼

評分結果(參考) ： **<font color="#00bb00">AC</font> (3ms, 332KB)**

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    while(cin >> n){
        vector<int> arr(n);
        for(int& i : arr) cin >> i;
        
        sort(arr.begin(), arr.end());

        for(int i=0; i<n; i++){
            if(i != 0) cout << " ";
            cout << arr[i];
        }
        cout << "\n";
    }
    
    return 0;
}
```
