## 解法一

### ✅ 完整代碼

評分結果(參考) ： **<font color="#00bb00">AC</font> (2ms, 336KB)**

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

signed main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, a;
    stack<int> stk;
    cin >> n;
    
    for(int i=0; i<n; i++){
        cin >> a;
        if(a == 1){
            stk.pop();
        }
        else if(a == 2){
            cout << stk.top() << "\n";
        }
        else if(a == 3){
            int num;
            cin >> num;
            stk.push(num);
        }
    }
    
    return 0;
}
```
