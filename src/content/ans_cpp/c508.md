## 解法一

### ✅ 完整代碼

評分結果(參考) ： **<font color="#00bb00">AC</font> (22ms, 1.1MB)**

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

signed main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    
    vector<int> a(n);
    for(int& i : a) cin >> i;
    
    sort(a.begin(), a.end());
    for(int i : a) cout << i << " ";
    cout << "\n";
    
    cout << a[n-1] << " ";
    for(int i=n-2; i>=0; i--){
        if(a[i] != a[i+1]) cout << a[i] << " ";
    }
    
    return 0;
}
```
