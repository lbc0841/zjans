#include <bits/stdc++.h>
using namespace std;
#define int long long

int f(int n){
    if(n == 1) return 1;
    else return (n + f(n-1));
}

int g(int n){
    if(n == 1) return 1;
    else return (f(n) + g(n-1));
}

signed main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int day;
    while(cin >> day){
        cout << f(day) << " " << g(day) << "\n";
    }

}
