## 解法一

### ✅ 完整代碼

評分結果(參考) ： **<font color="#00bb00">AC</font> (4ms, 1.8MB)**

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

signed main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    
    vector<int> a(n);
    for(int& i : a) cin >> i;
    
    vector<int> mx(n);
    mx[0] = a[0];
    for(int i=1; i<n; i++){
        mx[i] = max(mx[i-1], a[i]);
    }
    
    int ans = a[1]-a[0];
    for(int i=1; i<n; i++){
        ans = max(ans, mx[i-1]-a[i]);
    }
    cout << ans;
    
    return 0;
}
```
