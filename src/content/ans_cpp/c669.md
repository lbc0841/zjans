## 解法一

### ✅ 完整代碼

評分結果(參考) ： **<font color="#00bb00">AC</font> (1ms, 348KB)**

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

signed main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;
    cin.ignore(999,'\n');
    
    string line;
    for(int i=0; i<t; i++){
        getline(cin, line);
        stringstream ss(line);
        unordered_set<int> st;
        
        int num=0, sum=0, m=0, d=0, mx=0, mn=1e18;
        while(ss >> num){
            if(st.count(num)) d = num;
            else sum += num;
            
            mx = max(mx, num);
            mn = min(mn, num);
            st.insert(num);
        }
        m = (st.size()+1)*(mx+mn)/2 - sum;
        
        cout << m << " " << d << "\n";
    }
}
```
