#include <bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int m;
    cin >> m;

    if(25-m < 0) cout << 25-m + 60;
    else cout << 25-m;

    return 0;
}
