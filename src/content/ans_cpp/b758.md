#include<bits/stdc++.h>
#define int long long
using namespace std;

signed main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int x, y;
    cin >> x >> y;
    
    x += 2;
    y += 30;
    
    x += y/60;
    
    x %= 24;
    y %= 60;
    
    if(x <= 9) cout << "0";
    cout << x << ":";
    if(y <= 9) cout << "0";
    cout << y;
    
    return 0;
}
