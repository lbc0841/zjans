## 解法一、前綴和

### ✅ 完整代碼

評分結果(參考) ： **<font color="#00bb00">AC</font> (90ms, 928KB)**

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m, temp;
    while(cin >> n >> m){
        vector<int> presum(1, 0);
        for(int i=0; i<n; i++){
            cin >> temp;
            presum.push_back(presum.back()+temp);
        }
        
        for(int i=0; i<m; i++){
            int l, r;
            cin >> l >> r;
            cout << presum[r]-presum[l-1] << "\n";
        }
    }
    
    return 0;
}
```
