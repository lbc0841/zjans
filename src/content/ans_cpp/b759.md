```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

signed main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    string s;
    cin >> s;
    
    int l = s.size();
    s += s;
    
    for(int i=0; i<l; i++){
        for(int j=0; j<l; j++){
            cout << s[i+j];
        }
        cout << "\n";
    }
    
    return 0;
}
```

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

signed main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    string s;
    cin >> s;
    
    int l = s.size();
    s += s;
    
    for(int i=0; i<l; i++){
        for(int j=0; j<l; j++){
            putchar_unlocked(s[i+j]);
        }
        putchar_unlocked('\n');
    }
    
    return 0;
}
```
